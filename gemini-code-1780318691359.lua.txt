-- Players.EXDestructor01.PlayerGui.ScreenGui.MainPanel
local function C_1e()
	local script = LMG2L["MainPanel_16"] or LMG2L["MainPanel_1e"];
	
	task.spawn(function()	
		-------------------------------------------------------------------------
		-- SERVICES & BINDINGS
		-------------------------------------------------------------------------
		local TweenService = game:GetService("TweenService")	
		local MarketplaceService = game:GetService("MarketplaceService")
		local HttpService = game:GetService("HttpService")
		local Players = game:GetService("Players")
		local Selection = game:GetService("Selection") -- Digunakan jika berjalan di Studio/Plugin
		local LocalPlayer = Players.LocalPlayer
		local PlayerGui = LocalPlayer:WaitForChild("PlayerGui")
		
		-- Bindings Eksternal (Studio Lite Server Functions)
		local StudioLiteFolder = game:GetService("ReplicatedStorage"):WaitForChild("StudioLiteFolder", 5)
		local ServerFunctions = StudioLiteFolder and StudioLiteFolder:WaitForChild("ServerFunctions", 5)
		local LoadAssetRemote = StudioLiteFolder and StudioLiteFolder:WaitForChild("LoadAssetModelToPlayerGuiServerFunction", 5)
		local ClearAssetRemote = StudioLiteFolder and StudioLiteFolder:WaitForChild("ClearAssetModelToPlayerGuiServerFunction", 5)
		
		-- Bindings Explorer Panel bawaan Studio Lite (jika ada)
		local ExplorerPanel = script.Parent.Parent:FindFirstChild("ExplorerPanel")
		local SetSelectionRemote = ExplorerPanel and ExplorerPanel:FindFirstChild("SetSelection")
		local GetSelectionRemote = ExplorerPanel and ExplorerPanel:FindFirstChild("GetSelection")
		
		local writefile = writefile or io.writefile
		local readfile = readfile or io.readfile
		local isfile = isfile or io.isfile
		local makefolder = makefolder or io.makefolder
		local setclipboard = setclipboard or toclipboard or print

		-------------------------------------------------------------------------
		-- PEMANGGILAN OBJEK UI
		-------------------------------------------------------------------------
		local Gui = script.Parent	
		local MainPanel = Gui:WaitForChild("Panel")	
		local CloseButton = MainPanel:WaitForChild("CloseButton")	
		local PanahButton = Gui:WaitForChild("PanahButton")	
		
		local ModelButton = MainPanel:WaitForChild("ModelButton")
		local DecalButton = MainPanel:WaitForChild("DecalButton")
		local AudioButton = MainPanel:WaitForChild("AudioButton")
		
		local GetAssetIdBox = MainPanel:WaitForChild("GetAssetId")
		local GetButton = GetAssetIdBox:WaitForChild("GetButton")
		
		local AssetIdBox = MainPanel:WaitForChild("AssetId")
		local SaveButton = AssetIdBox:WaitForChild("SaveButton")
		
		local ScrollingFrame = MainPanel:WaitForChild("ScrollingFrame")
		local TemplateFrame = ScrollingFrame:WaitForChild("Frame")
		
		TemplateFrame.Visible = false 
		TemplateFrame.Parent = nil

		-------------------------------------------------------------------------
		-- CONFIGURATION & DATA SAVED STATE
		-------------------------------------------------------------------------
		local CurrentCategory = "Model" 
		local SavedAssets = {
			Model = {89464989224212, 16063473188, 8222709011},
			Decal = {4846381420, 6914010536},
			Audio = {118149279616179, 124112959171614, 5801257793}
		}

		local COLOR_ACTIVE = Color3.fromRGB(29, 171, 223)   
		local COLOR_INACTIVE = Color3.fromRGB(36, 36, 36) 
		local Debounce = false

		if makefolder and isfile and readfile then
			pcall(function()
				makefolder("delta")
				if isfile("delta/toolbox_assets.json") then
					local decoded = HttpService:JSONDecode(readfile("delta/toolbox_assets.json"))
					if decoded then SavedAssets = decoded end
				end
			end)
		end

		local function SaveData()
			if writefile then
				pcall(function()
					writefile("delta/toolbox_assets.json", HttpService:JSONEncode(SavedAssets))
				end)
			end
		end

		local function ClearList()
			for _, item in ipairs(ScrollingFrame:GetChildren()) do
				if item:IsA("Frame") then item:Destroy() end
			end
		end

		-- Mengambil item aktif yang sedang dipilih oleh user di runtime hierarchy
		local function GetCurrentSelection()
			if GetSelectionRemote then
				local sel = GetSelectionRemote:Invoke()
				if sel and typeof(sel) == "table" and #sel >= 1 then return sel[1] end
			elseif Selection then
				local sel = Selection:Get()
				if sel and #sel >= 1 then return sel[1] end
			end
			return nil
		end

		-- Memaksa UI Explorer kustom untuk menyeleksi objek baru
		local function SetCurrentSelection(obj)
			if SetSelectionRemote then pcall(function() SetSelectionRemote:Invoke({obj}) end) end
		end

		-------------------------------------------------------------------------
		-- SISTEM INSERT UTAMA (WORKSPACE, LIGHTING, PART SELECTION, SOUNDSERVICE)
		-------------------------------------------------------------------------
		local function InsertAsset(assetId, category, statusTarget)
			if Debounce then return end
			Debounce = true
			statusTarget.Text = "Working"
			
			local stringId = tostring(assetId)
			local assetInfo = nil
			
			-- Deteksi Kategori Utama via Marketplace jika parameter kosong
			local success, info = pcall(function() return MarketplaceService:GetProductInfo(assetId) end)
			if success and info then 
				assetInfo = info
				if not category then
					if info.AssetTypeId == 13 or info.AssetTypeId == 3 or info.AssetTypeId == 14 then category = "Decal"
					elseif info.AssetTypeId == 34 then category = "Audio"
					else category = "Model" end
				end
			else
				category = category or "Model"
			end

			local targetInsertedObject = nil
			local currentSelectedInstance = GetCurrentSelection()

			-----------------------------------------------------------------
			-- LOGIKA 1: AUDIO/SOUND SYSTEM
			-----------------------------------------------------------------
			if category == "Audio" then
				local sound = Instance.new("Sound")
				sound.Name = (assetInfo and assetInfo.Name) or "Sound_" .. stringId
				sound.SoundId = "rbxassetid://" .. stringId
				sound.Volume = 0.5
				
				-- Jika user memilih objek di workspace, sound dimasukkan ke dalam objek tersebut (3D Spatial Sound)
				if currentSelectedInstance and (currentSelectedInstance:IsA("BasePart") or currentSelectedInstance:IsA("Attachment")) then
					sound.Parent = currentSelectedInstance
				else
					sound.Parent = game:GetService("SoundService")
				end
				
				sound:Play()
				targetInsertedObject = sound
				statusTarget.Text = "Berhasil!"

			-----------------------------------------------------------------
			-- LOGIKA 2: DECAL/IMAGE SYSTEM
			-----------------------------------------------------------------
			elseif category == "Decal" then
				local decal = Instance.new("Decal")
				decal.Name = (assetInfo and assetInfo.Name) or "Decal_" .. stringId
				decal.Texture = "rbxassetid://" .. stringId
				
				-- Jika ada Part terpilih, Decal langsung dipasang di permukaannya
				if currentSelectedInstance and currentSelectedInstance:IsA("BasePart") then
					decal.Parent = currentSelectedInstance
				else
					-- Fallback membuat part dummy agar decal bisa terlihat di Workspace
					local containerPart = Instance.new("Part")
					containerPart.Name = "DecalContainer_" .. stringId
					containerPart.Size = Vector3.new(4, 4, 1)
					containerPart.Position = workspace.Camera.CFrame.Position + (workspace.Camera.CFrame.LookVector * 10)
					decal.Parent = containerPart
					containerPart.Parent = workspace
					targetInsertedObject = containerPart
				end
				if not targetInsertedObject then targetInsertedObject = decal end
				statusTarget.Text = "Berhasil!"

			-----------------------------------------------------------------
			-- LOGIKA 3: MODEL / SKYBOX SYSTEM (STUDIO LITE BINDINGS & FALLBACK)
			-----------------------------------------------------------------
			else
				-- Jika menggunakan Remote Studio Lite Server Functions / Custom Framework
				if ServerFunctions then
					pcall(function()
						targetInsertedObject = ServerFunctions:InvokeServer("LoadAssetToPlayerGui", stringId)
					end)
				end

				if not targetInsertedObject and LoadAssetRemote then
					local loadSuccess = false
					pcall(function() loadSuccess = LoadAssetRemote:InvokeServer(stringId) end)
					if loadSuccess then targetInsertedObject = PlayerGui:WaitForChild(stringId, 5) end
				end

				-- Eksekusi Parsing Object dari Transit Storage (ServerFolder / PlayerGui)
				if targetInsertedObject then
					local assetClone = targetInsertedObject:Clone()
					targetInsertedObject:Destroy()
					
					-- Bersihkan sisa cache di server side jika menggunakan native method
					if ClearAssetRemote then ClearAssetRemote:InvokeServer(stringId) end
					if ServerFunctions then ServerFunctions:InvokeServer("ClearAssetFromPlayerGui", stringId) end
					
					local mainInstance = assetClone:GetChildren()[1]
					if mainInstance then
						-- Cek jika objek bermigrasi ke tempat khusus seperti Lighting (Skybox / PostEffect)
						if mainInstance:IsA("Sky") or mainInstance:IsA("PostEffect") or mainInstance.Name:lower():find("sky") then
							mainInstance.Parent = game:GetService("Lighting")
							targetInsertedObject = mainInstance
						else
							-- Logika Spasial Raycast CFrame Penempatan Objek Fisik di Workspace
							local targetModel
							local isTempWrapper = false
							
							if mainInstance:IsA("Model") then
								targetModel = mainInstance
							else
								targetModel = Instance.new("Model")
								mainInstance.Parent = targetModel
								isTempWrapper = true
							end
							
							local boundingCFrame, boundingSize = targetModel:GetBoundingBox()
							local lowestYOffset = not targetModel.PrimaryPart and 0 or targetModel.PrimaryPart.Position.Y - boundingSize.Y / 2
							
							local camera = workspace.Camera
							local camCFrame = camera.CFrame
							local posX = math.floor((camCFrame.X + camCFrame.LookVector.X * 25) * 2) / 2
							local posY = boundingSize.Y / 2 + lowestYOffset
							local posZ = math.floor((camCFrame.Z + camCFrame.LookVector.Z * 25) * 2) / 2
							
							local targetPos = Vector3.new(posX, posY, posZ)
							local raycastResult = workspace:Raycast(Vector3.new(targetPos.X, camCFrame.Y, targetPos.Z), Vector3.new(0, -camCFrame.Y * 2, 0))
							
							if raycastResult then
								local newY = raycastResult.Instance.Position.Y + (raycastResult.Instance.Size.Y / 2) + (boundingSize.Y / 2) + lowestYOffset
								targetPos = Vector3.new(targetPos.X, newY, targetPos.Z)
							end
							
							targetModel:PivotTo(CFrame.new(targetPos) * boundingCFrame.Rotation)
							
							if isTempWrapper then
								local finalRawObject = targetModel:GetChildren()[1]:Clone()
								finalRawObject.Parent = workspace
								targetInsertedObject = finalRawObject
								targetModel:Destroy()
							else
								targetModel.Parent = workspace
								targetInsertedObject = targetModel
							end
						end
						statusTarget.Text = "Berhasil!"
					end
					assetClone:Destroy()
				else
					-- Executor Client Fallback (GetObjects) jika Studio Lite Environment sedang mati
					local clientSuccess, clientObj = pcall(function() return game:GetObjects("rbxassetid://" .. assetId)[1] end)
					if clientSuccess and clientObj then
						local clone = clientObj:Clone()
						if clone:IsA("Sky") or clone:IsA("PostEffect") then
							clone.Parent = game:GetService("Lighting")
						else
							clone.Parent = workspace
							if clone:IsA("Model") then
								clone:MakeJoints()
								if LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart") then
									clone:MoveTo(LocalPlayer.Character.HumanoidRootPart.Position + Vector3.new(0, 4, 0))
								end
							end
						end
						targetInsertedObject = clone
						statusTarget.Text = "Berhasil!"
					else
						statusTarget.Text = "Gagal"
					end
				end
			end

			-- Seleksi Otomatis Objek Baru di Editor Explorer
			if targetInsertedObject then SetCurrentSelection(targetInsertedObject) end
			
			task.wait(1)
			Debounce = false
		end

		-------------------------------------------------------------------------
		-- RENDER LIST ASSET & ADJUST CANVAS
		-------------------------------------------------------------------------
		local function RenderAssets()
			ClearList()
			local targetList = SavedAssets[CurrentCategory] or {}

			local function UpdateCanvas()
				local layout = ScrollingFrame:FindFirstChild("UIListLayout")
				if layout then
					ScrollingFrame.CanvasSize = UDim2.new(0, 0, 0, layout.AbsoluteContentSize.Y + 15)
				end
			end

			for _, assetId in ipairs(targetList) do
				task.spawn(function()
					local success, info = pcall(function() return MarketplaceService:GetProductInfo(assetId) end)
					if success and info then
						local card = TemplateFrame:Clone()
						card.Visible = true
						card.Parent = ScrollingFrame
						card.Name = "Asset_" .. assetId

						local SalinButton = card:WaitForChild("SalinButton")
						local InsertButton = card:WaitForChild("InsertButton")
						local ImageAsset = card:WaitForChild("ImageAsset")
						local IdAsset = card:WaitForChild("IdAsset")
						local ByPemilikAsset = card:WaitForChild("ByPemilikAsset")
						local NameAsset = card:WaitForChild("NameAsset")

						NameAsset.TextWrapped = true
						NameAsset.TextYAlignment = Enum.TextYAlignment.Top
						NameAsset.Text = info.Name
						
						ByPemilikAsset.Text = "By: " .. (info.Creator and info.Creator.Name or "Unknown")
						IdAsset.Text = "ID: " .. tostring(assetId)

						-- Konfigurasi Tampilan Thumbnail Sesuai Kategori Aturan Komponen
						if CurrentCategory == "Decal" then
							ImageAsset.Image = "rbxthumb://type=Asset&id=" .. assetId .. "&w=150&h=150"
						elseif CurrentCategory == "Audio" then
							ImageAsset.Image = "rbxassetid://16327318049" -- Default Audio Vinyl Icon
						else
							ImageAsset.Image = "rbxthumb://type=Asset&id=" .. assetId .. "&w=150&h=150"
						end

						SalinButton.MouseButton1Click:Connect(function()
							setclipboard(tostring(assetId))
							SalinButton.Text = "Tersalin!"
							task.wait(1)
							SalinButton.Text = "Salin"
						end)

						InsertButton.MouseButton1Click:Connect(function()
							InsertAsset(assetId, CurrentCategory, InsertButton)
							task.wait(1)
							InsertButton.Text = "Insert"
						end)

						UpdateCanvas()
					end
				end)
			end
			
			task.defer(function()
				for i = 1, 6 do UpdateCanvas() task.wait(0.05) end
			end)
		end

		local function SwitchTab(tabName)
			CurrentCategory = tabName
			ModelButton.BackgroundColor3 = COLOR_INACTIVE
			DecalButton.BackgroundColor3 = COLOR_INACTIVE
			AudioButton.BackgroundColor3 = COLOR_INACTIVE

			if tabName == "Model" then ModelButton.BackgroundColor3 = COLOR_ACTIVE
			elseif tabName == "Decal" then DecalButton.BackgroundColor3 = COLOR_ACTIVE
			elseif tabName == "Audio" then AudioButton.BackgroundColor3 = COLOR_ACTIVE end
			RenderAssets()
		end

		-------------------------------------------------------------------------
		-- ACTION LISTENERS (TAB, GET & VALIDASI PENYIMPANAN SAVE)
		-------------------------------------------------------------------------
		ModelButton.MouseButton1Click:Connect(function() SwitchTab("Model") end)
		DecalButton.MouseButton1Click:Connect(function() SwitchTab("Decal") end)
		AudioButton.MouseButton1Click:Connect(function() SwitchTab("Audio") end)

		-- PENCARIAN GANDA: Mendapatkan info Meta, Menampilkan Info ke Box, lalu Auto-Insert
		GetButton.MouseButton1Click:Connect(function()
			local cleanId = tonumber(GetAssetIdBox.Text:match("%d+"))
			if not cleanId then
				GetAssetIdBox.Text = "Harus ID Angka!"
				task.wait(1.5)
				GetAssetIdBox.Text = "Masukan ID Asset .."
				return
			end

			GetButton.Text = "LOAD"
			local success, info = pcall(function() return MarketplaceService:GetProductInfo(cleanId) end)
			
			if success and info then
				-- Tampilkan info nama aset yang ditemukan ke dalam Box Pencarian sebagai Feedback pencarian
				GetAssetIdBox.Text = "[" .. info.Name .. "]"
				
				-- Proses Instancing Pemasukan Aset
				InsertAsset(cleanId, nil, GetButton)
			else
				GetAssetIdBox.Text = "ID Tidak Valid!"
			end
			
			task.wait(2)
			GetButton.Text = "GET"
			GetAssetIdBox.Text = "Masukan ID Asset .."
		end)

		-- SISTEM SAVE AKURAT: Validasi Kategori Tepat & Anti Duplikasi Silang Kategori
		SaveButton.MouseButton1Click:Connect(function()
			local cleanId = tonumber(AssetIdBox.Text:match("%d+"))
			if not cleanId then
				AssetIdBox.Text = "Harus ID Angka!"
				task.wait(1.5)
				AssetIdBox.Text = "Masukan ID Asset.."
				return
			end

			SaveButton.Text = "CHECKING"
			local success, info = pcall(function() return MarketplaceService:GetProductInfo(cleanId) end)

			if success and info then
				-- Klasifikasi Kategori Target Berdasarkan System Atribut Atribut API Resmi Roblox
				local detectedCat = "Model"
				if info.AssetTypeId == 13 or info.AssetTypeId == 3 or info.AssetTypeId == 14 then 
					detectedCat = "Decal"
				elseif info.AssetTypeId == 34 then 
					detectedCat = "Audio" 
				end

				-- Periksa Apakah ID Sudah Terdaftar di Kategori Manapun untuk Mencegah Redundansi Data
				local isDuplicated = false
				for _, checkId in ipairs(SavedAssets.Model) do if checkId == cleanId then isDuplicated = true break end end
				for _, checkId in ipairs(SavedAssets.Decal) do if checkId == cleanId then isDuplicated = true break end end
				for _, checkId in ipairs(SavedAssets.Audio) do if checkId == cleanId then isDuplicated = true break end end

				if not isDuplicated then
					table.insert(SavedAssets[detectedCat], cleanId)
					SaveData()
					SwitchTab(detectedCat) -- Auto-fokus beralih ke tab kategori item yang baru tersimpan
					AssetIdBox.Text = "Tersimpan!"
				else
					AssetIdBox.Text = "Sudah Ada!"
				end
			else
				AssetIdBox.Text = "ID Gagal Validasi!"
			end
			task.wait(2)
			SaveButton.Text = "SAVE"
			AssetIdBox.Text = "Masukan ID Asset.."
		end)

		-------------------------------------------------------------------------
		-- PANEL NAVIGASI POSITIONING & FADE EFFECT CLOSING
		-------------------------------------------------------------------------
		local OpenTween = TweenInfo.new(0.35, Enum.EasingStyle.Quart, Enum.EasingDirection.Out)	
		local CloseTween = TweenInfo.new(0.3, Enum.EasingStyle.Quart, Enum.EasingDirection.In)	
	
		local PANEL_OPEN = UDim2.new(0, 5, 0, 20)	
		local PANEL_HIDE = UDim2.new(0, -270, 0, 20)	
		
		local ARROW_OPEN = UDim2.new(0, 270, 0, 152)	
		local ARROW_HIDE = UDim2.new(0, 5, 0, 152)	
	
		local Hidden = false	
		local Busy = false	
	
		local function TogglePanel()	
			if Busy then return end	
			Busy = true	
	
			if not Hidden then	
				Hidden = true	
				PanahButton.Text = ">"	
				TweenService:Create(MainPanel, OpenTween, {Position = PANEL_HIDE}):Play()	
				TweenService:Create(PanahButton, OpenTween, {Position = ARROW_HIDE}):Play()	
			else	
				Hidden = false	
				PanahButton.Text = "<"	
				TweenService:Create(MainPanel, OpenTween, {Position = PANEL_OPEN}):Play()	
				TweenService:Create(PanahButton, OpenTween, {Position = ARROW_OPEN}):Play()	
			end	
			task.wait(0.35)	
			Busy = false
		end	
	
		local function CloseGui()	
			if Busy then return end	
			Busy = true	
			TweenService:Create(MainPanel, CloseTween, {Size = UDim2.new(0, 0, 0, 0), BackgroundTransparency = 1}):Play()	
			TweenService:Create(PanahButton, CloseTween, {TextTransparency = 1, BackgroundTransparency = 1}):Play()	
			for _, v in ipairs(MainPanel:GetDescendants()) do	
				if v:IsA("TextLabel") or v:IsA("TextButton") or v:IsA("TextBox") then	
					TweenService:Create(v, CloseTween, {TextTransparency = 1, BackgroundTransparency = 1}):Play()	
				elseif v:IsA("ImageLabel") or v:IsA("ImageButton") then	
					TweenService:Create(v, CloseTween, {ImageTransparency = 1, BackgroundTransparency = 1}):Play()	
				elseif v:IsA("Frame") or v:IsA("ScrollingFrame") then	
					TweenService:Create(v, CloseTween, {BackgroundTransparency = 1}):Play()	
				end	
			end	
			task.wait(0.35)	
			if Gui then Gui:Destroy() end	
		end	
	
		PanahButton.MouseButton1Click:Connect(TogglePanel)	
		CloseButton.MouseButton1Click:Connect(CloseGui)	

		SwitchTab("Model")
	end)	
end;
task.spawn(C_1e);

return LMG2L["ScreenGui_1"], require;
